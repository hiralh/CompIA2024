    //When a student's name is clicked on the studentList,
    private void studentListValueChanged(javax.swing.event.ListSelectionEvent evt) {                                         
        if (!evt.getValueIsAdjusting()) {
            //retrieve name of selected student
            selectedStudentName = studentList.getSelectedValue();
            if (selectedStudentName != null && !selectedStudentName.isEmpty()) {
                try {
                    //create an SQL query to retrieve the names of all the tasks of the selected student
                    String query = "SELECT taskName FROM tasks WHERE sName=?";
                    try (PreparedStatement pstmt = conn.prepareStatement(query)) {
                        pstmt.setString(1, selectedStudentName);
                        try (ResultSet rs = pstmt.executeQuery()) {
                            //taskList model cleared in case any task names were being displayed
                            tasMod.clear();
                            //all task names of the selected student are added to tasMod and displayed
                            while (rs.next()) {
                                String taskName = rs.getString("taskName");
                                tasMod.addElement(taskName);
                                }
                        }
                    }

                } catch (SQLException e) {
                    JOptionPane.showMessageDialog(this, 
                            "Failed to load tasks for the selected student.", "Error", 
                            JOptionPane.ERROR_MESSAGE);
                }
            }
        }
    }                                        

    //When the view button is clicked,
    private void viewListActionPerformed(java.awt.event.ActionEvent evt) {                                         
        //student list model is cleared in case an names were being displayed
        stuMod.clear();
        try {
            try (Statement stmt = conn.createStatement()) {
                //retrieving names of all student currently registered
                ResultSet rs = stmt.executeQuery("SELECT sName FROM studentInfo");
                //adding all the retrieved student names to stuMod
                while (rs.next()) {
                    String studentName = rs.getString("sName");
                    stuMod.addElement(studentName);
                }
            }
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Failed to load student names.", 
                    "Error", JOptionPane.ERROR_MESSAGE);
        }
    }                                        

    //When - button pressed,
    private void deleteActionPerformed(java.awt.event.ActionEvent evt) {                                       
        //enter name of the task to be deleted
        selectedTaskName = JOptionPane.showInputDialog("Enter the name of the task:");
        if (selectedTaskName != null && !selectedTaskName.isEmpty()) {
            try {
                //create a SQL query to select all columns of data for the selected student name, specifically for the entered task name
                String query = "SELECT * FROM tasks WHERE sName = ? AND taskName= ?";
                PreparedStatement pstmt = conn.prepareStatement(query);
                pstmt.setString(1, selectedStudentName);
                pstmt.setString(2, selectedTaskName);
                ResultSet rs = pstmt.executeQuery();
                if(rs.next()){
                    //deleting the entire record of the entered task name for the selected student
                    query = "DELETE FROM tasks WHERE taskName= ? AND sName= ?";
                    pstmt = conn.prepareStatement(query);
                    pstmt.setString(1, selectedTaskName);
                    pstmt.setString(2, selectedStudentName);
                    pstmt.executeUpdate();
                    
                    //remove task name from the list model called 'tasMod' for the selected student
                    JOptionPane.showMessageDialog(this, "Task '" + selectedTaskName + "' deleted successfully.");
                    tasMod.removeElement(selectedTaskName);
                }
                else {
                JOptionPane.showMessageDialog(this, "No task found with the name '" + selectedTaskName + "'.");
            }
            } catch (SQLException ex) {
                Logger.getLogger(studentInfo.class.getName()).log(Level.SEVERE, null, ex);
            }
        }
    }                                      

    //When the + button is pressed,
    private void addActionPerformed(java.awt.event.ActionEvent evt) {                                    
        //enter name of task to be added
        String newTaskName = JOptionPane.showInputDialog("Enter the name of new task:");
        if (newTaskName != null && !newTaskName.isEmpty()) {
            try {
                //create a SQL query to create a task record for the selected student with the entered task name 
                String query = "INSERT INTO tasks (sName, taskName) VALUES (?, ?)";
                PreparedStatement pstmt = conn.prepareStatement(query);
                        pstmt.setString(1, selectedStudentName);
                        pstmt.setString(2, newTaskName);
                        pstmt.executeUpdate();
                        
                        //add task name in the list model called 'tasMod' for the selected student
                        tasMod.addElement(newTaskName);
            } catch (SQLException ex) {
                Logger.getLogger(studentInfo.class.getName()).log(Level.SEVERE, null, ex);
            }
        }
    }

    //When a task name is selected from the task list of the selected student
    private void taskListValueChanged(javax.swing.event.ListSelectionEvent evt) {                                      
        if (!evt.getValueIsAdjusting()) {
            //retrieve name of selected task from taskList
            selectedTaskName = taskList.getSelectedValue();
            //open task information editor page for the selected task
            openTaskEditor(selectedTaskName);
        }
    }                                     

    private void openTaskEditor(String taskName) {
    if (taskName != null && !taskName.isEmpty()) {
        try {
            //retrieving current task information for selected task from the task list of the selected student
            String query = "SELECT taskName, dueDate, completed, dueHour, dueMin FROM tasks WHERE sName=? AND taskName=?";
            try (PreparedStatement pstmt = conn.prepareStatement(query)) {
                pstmt.setString(1, selectedStudentName);
                pstmt.setString(2, taskName);
                try (ResultSet rs = pstmt.executeQuery()) {
                    if (rs.next()) {
                        tasEditor = new taskEditor(conn);
                        this.setVisible(false); //tasks page closed
                        tasEditor.setVisible(true); //task information editor page opened
                        //setting task information editor with currently retrieved details
                        tasEditor.setTaskDetails(rs.getString("taskName"), 
                                rs.getString("dueDate"), rs.getInt("completed"), 
                                rs.getString("dueHour"), rs.getString("dueMin"));
                    } else {
                        JOptionPane.showMessageDialog(this, "No task found with the name '" + taskName + "'.");
                    }
                }
            }
        }catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Failed to load task details.", "Error", 
                    JOptionPane.ERROR_MESSAGE);
            }
        }
    }
