    //When student name is clicked on student list,
    private void studentListValueChanged(javax.swing.event.ListSelectionEvent evt) {                                         
        if (!evt.getValueIsAdjusting()) {
            //retrieve selected student's name
            selectedStudentName = studentList.getSelectedValue();
            if (selectedStudentName != null && !selectedStudentName.isEmpty()) {
                try {
                    //create a SQL query to retrieve names of all the fee records for the selected student
                    String query = "SELECT feeName FROM fees WHERE sName=?";
                    try (PreparedStatement pstmt = conn.prepareStatement(query)) {
                        pstmt.setString(1, selectedStudentName);
                        try (ResultSet rs = pstmt.executeQuery()) {
                            //clear fee list model in case any fee names were being displayed
                            feeMod.clear();
                            //display all fee record names for selected student on fee list 
                            while (rs.next()) {
                                String feeName = rs.getString("feeName");
                                //adding fee record names to fee list model
                                feeMod.addElement(feeName);
                            }
                        }
                    }

                } catch (SQLException e) {
                    JOptionPane.showMessageDialog(this, 
                            "Failed to load files for the selected student.", "Error", JOptionPane.ERROR_MESSAGE);
                }
            }
        }
    }                                        

    //When view button is pressed,
    private void viewListActionPerformed(java.awt.event.ActionEvent evt) {                                         
        //clear student model list in case any student names were being displayed
        stuMod.clear();
        
        try {
            try (Statement stmt = conn.createStatement()) {
                //retrieve all currently registered student names
                ResultSet rs = stmt.executeQuery("SELECT sName FROM studentInfo");

                //display all currently registered student names in the student list
                while (rs.next()) {
                    String studentName = rs.getString("sName");
                    //add student name to student list model
                    stuMod.addElement(studentName);
                }
            }
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Failed to load student names.", "Error", JOptionPane.ERROR_MESSAGE);
        }
    }                                        

    //When - button is pressed,
    private void deleteActionPerformed(java.awt.event.ActionEvent evt) {                                       
        //enter name of fee record to be deleted
        selectedFeeName = JOptionPane.showInputDialog("Enter the name of the fee record:");
        if (selectedFeeName != null && !selectedFeeName.isEmpty()) {
            try {
                //select all columns of data of the user-entered fee name for the selected sudent
                String query = "SELECT * FROM fees WHERE sName = ? AND feeName= ?";
                PreparedStatement pstmt = conn.prepareStatement(query);
                pstmt.setString(1, selectedStudentName);
                pstmt.setString(2, selectedFeeName);
                ResultSet rs = pstmt.executeQuery();
                if(rs.next()){
                    //delete record for the user-entered fee name of the selected student
                    query = "DELETE FROM fees WHERE feeName= ? AND sName= ?";
                    pstmt = conn.prepareStatement(query);
                    pstmt.setString(1, selectedFeeName);
                    pstmt.setString(2, selectedStudentName);
                    pstmt.executeUpdate();

                    JOptionPane.showMessageDialog(this, "Fee record '" + selectedFeeName +"' deleted successfully.");
                }
                else {
                JOptionPane.showMessageDialog(this, "No fee record found with the name '" +selectedFeeName + "'.");
                }
            } catch (SQLException ex) {
                Logger.getLogger(studentInfo.class.getName()).log(Level.SEVERE, null, ex);
            }
        }
    }                                      

    //When the + action is pressed,
    private void addActionPerformed(java.awt.event.ActionEvent evt) {                                    
        //enter name of fee record to be added
        String newFeeName = JOptionPane.showInputDialog("Enter the name of new fee record:");
        if (newFeeName != null && !newFeeName.isEmpty()) {
            try {
                //create a SQL query to create a record of the user-entered fee name for the selected student
                String query = "INSERT INTO fees (sName, feeName) VALUES (?, ?)";
                PreparedStatement pstmt = conn.prepareStatement(query);
                pstmt.setString(1, selectedStudentName);
                pstmt.setString(2, newFeeName);
                pstmt.executeUpdate();

                //add fee record name to fee list model
                feeMod.addElement(newFeeName);
            } catch (SQLException ex) {
                Logger.getLogger(studentInfo.class.getName()).log(Level.SEVERE, null, ex);
            }
        }
    }

    //When fee name selected from fee list,
    private void feeListValueChanged(javax.swing.event.ListSelectionEvent evt) {                                     
        if (!evt.getValueIsAdjusting()) {
            //retrieve selected fee record name
            selectedFeeName = feeList.getSelectedValue();
            //open fee editor for selected fee record
            openFeeEditor(selectedFeeName);
        }
    }                                    

    private void openFeeEditor(String feeName) {
    if (feeName != null && !feeName.isEmpty()) {
        try {
            //create a SQL query to select current details for the selected fee record of the selected student
            String query = "SELECT feeName, totalAmount, feePaid, startDate, endDate, paid FROM fees WHERE sName=? AND feeName=?";
            try (PreparedStatement pstmt = conn.prepareStatement(query)) {
                pstmt.setString(1, selectedStudentName);
                pstmt.setString(2, feeName);
                try (ResultSet rs = pstmt.executeQuery()) {
                    
                    if (rs.next()) {
                        feeEditor = new feeEditor(conn);
                        this.setVisible(false); //fees page closed
                        feeEditor.setVisible(true); //fee editor page shown
                        //calling method from fee editor page to set fee details into the JTextFields of the fee editor page
                        feeEditor.setFeeDetails(rs.getString("feeName"),
                                rs.getDouble("totalAmount"), 
                                rs.getDouble("feePaid"), 
                                rs.getString("startDate"),
                                rs.getString("endDate"), rs.getInt("paid") );
                    } else {
                        JOptionPane.showMessageDialog(this, "No fee record found with the name '" + feeName + "'.");
                    }
                }
            }
        }catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Failed to load fee details.", "Error", 
                    JOptionPane.ERROR_MESSAGE);
            }
        }
    }
