    //if login button is pressed
    private void loginActionPerformed(java.awt.event.ActionEvent evt) {                                      
        JOptionPane.showMessageDialog(login, "WELCOME TO EDUTRACK 2.0");
        String E= null, P= null;
        PreparedStatement pstmt= null;
        ResultSet rs = null;
        try {          
            E= email.getText(); //extracting the email that user has entered JTextField
            P= password.getText(); //extracting the password that user has entered in JTextField
            //creating a SQL query to extract all columns of data which have a certain email and password
            pstmt = conn.prepareStatement("SELECT * FROM login WHERE email = ? AND password = ?");
            //assigning parameters (starting from 1 because index in SQL starts from 1 not 0
            pstmt.setString(1, E);
            pstmt.setString(2, P);
            //storing results of the above SQL query in a variable rs of type ResultSet
            rs= pstmt.executeQuery();
            if(rs.next()) //if rs has at least one row of results
            {
                this.setVisible(false); //login page is hidden
                dash = new dashboard(conn);
                dash.setVisible(true); //dashboard page is made visible
            }
            else{
                JOptionPane.showMessageDialog(login, "Incorrect email or password.");
            }
            }catch (SQLException ex) {
                Logger.getLogger(login.class.getName()).log(Level.SEVERE, null, ex);
            }
    }
