    //When the delete button is pressed,
    private void exitActionPerformed(java.awt.event.ActionEvent evt) {                                     
        JOptionPane.showMessageDialog(this, "Delete edits?");
        this.setVisible(false); //file editor oage closed
        fil= new files(conn);
        fil.setVisible(true); //files page opened
    }                                    

    //When the done button is pressed,
    private void DONEActionPerformed(java.awt.event.ActionEvent evt) {                                     
        //retrieving currently entered data from the JTextFields
        String fName = fileName.getText();
        String fPath = fileSpace.getText();
        try{
            //create a SQL query to update the database with the text entered in the JTextFields
            String query= "UPDATE files SET fileName=?, file=? WHERE sName=? AND fileName=?";
            try (
                PreparedStatement pstmt = conn.prepareStatement(query)) {
                pstmt.setString(1, fName);
                pstmt.setString(2, fPath);
                pstmt.setString(3, selectedStudentName);
                pstmt.setString(4, selectedFileName);
                pstmt.executeUpdate();
                
                JOptionPane.showMessageDialog(this, "Details have been updated");
            } catch (SQLException ex) {
                Logger.getLogger(fileEditor.class.getName()).log(Level.SEVERE, null, ex);
            }
            this.setVisible(false); //close file editor page
            fil= new files(conn);
            fil.setVisible(true); //files page shown
        }catch (HeadlessException e) {
            e.printStackTrace();
        }
    }                                    

    //When choose file button is pressed,
    private void chooseButtonActionPerformed(java.awt.event.ActionEvent evt) {                                             
        JFileChooser chooseFile= new JFileChooser();
        
        //display the file chooser dialog to the user in the centre of the screen
        chooseFile.showOpenDialog(null);
        
        //returning a File object 'f' representing the selected file.
        File f= chooseFile.getSelectedFile();
        
        //retrieve the absolute path of the selected file.
        filePath= f.getAbsolutePath();
        fileSpace.setText(filePath);
    }                                            

    public void setFileDetails(String fileName, String filePath)
    {
        //setting details into JTextFields in the file editor page
        if (fileName != null) {
            this.fileName.setText(fileName);
        } else {
            this.fileName.setText("");
        }
        
        if (filePath != null) {
            this.fileSpace.setText(filePath);
        } else {
            this.fileSpace.setText("");
        }
    }
